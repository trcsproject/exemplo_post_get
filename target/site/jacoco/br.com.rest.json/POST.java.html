package br.com.rest.json;

import com.fasterxml.jackson.annotation.JsonInclude.Include;
import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import org.json.JSONObject;
import org.apache.http.HttpResponse;
import org.apache.http.HttpStatus;
import org.apache.http.client.HttpClient;
import org.apache.http.client.methods.HttpPost;
import org.apache.http.entity.StringEntity;
import org.apache.http.impl.client.DefaultHttpClient;
import org.apache.http.protocol.HTTP;
import org.json.JSONException;

public class POST {

    public static void main(String[] args) throws JsonProcessingException, JSONException {

        JSONObject jsonCustomerContact = new JSONObject();
        JSONObject jsonContact = new JSONObject();

        jsonCustomerContact.put("contact", jsonContact);

        ObjectMapper mapper = new ObjectMapper();
        mapper.setSerializationInclusion(Include.NON_NULL);

        String jsonInString = mapper.writeValueAsString(jsonCustomerContact);

        HttpClient httpClient = new DefaultHttpClient();

        try {
            HttpPost request = new HttpPost("");
            StringEntity params = new StringEntity(jsonInString, HTTP.UTF_8);
            request.addHeader("content-type", "application/json");
            request.addHeader("Accept", "application/json");
            request.setEntity(params);
            HttpResponse response = httpClient.execute(request);

            if (response != null && response.getStatusLine() != null && response.getStatusLine().getStatusCode() == HttpStatus.SC_OK) {
                System.out.println("OK");
            }
        } catch (Exception ex) {
            System.out.println("Falha - " + ex.getMessage());
        } finally {
            httpClient.getConnectionManager().shutdown();
        }
    }

}
